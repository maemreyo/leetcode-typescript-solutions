# 1189. Maximum Number Of Balloons

### O(n) - TS with hash map

## Complexity

- Time: $$O(n)$$
- Space: $$O(n)$$

## Submission

Runtime: 2 ms · Beats 96.70%

## Code

```typescript
function maxNumberOfBalloons(text: string): number {
  const chars = text.replace(/[^balon]+/g, "").split("");
  let res = 0;

  const hash: Record<string, number> = {
    b: 0,
    a: 0,
    l: 0,
    o: 0,
    n: 0,
  };

  chars.forEach((char) => {
    hash[char]++;
  });

  while (hash.b > 0 && hash.a > 0 && hash.l > 1 && hash.o > 1 && hash.n > 0) {
    res++;
    hash.b--;
    hash.a--;
    hash.l -= 2;
    hash.o -= 2;
    hash.n--;
  }

  return res;
}
```
