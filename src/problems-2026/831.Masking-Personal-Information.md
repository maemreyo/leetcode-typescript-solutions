# 831. Masking Personal Information

```typescript
const getEmailMask = (e: string) => {
  const [name, domain] = e.toLowerCase().split("@");
  return `${name[0].at(0)}*****${name.at(-1)}@${domain}`;
};

const getPhoneMask = (p: string) => {
  const digits = p.replace(/\D/g, "");
  let start = "";

  switch (digits.length) {
    case 11:
      start = "+*-";
      break;
    case 12:
      start = "+**-";
      break;
    case 13:
      start = "+***-";
      break;
    default:
      break;
  }

  return `${start}***-***-${digits.slice(-4)}`;
};

function maskPII(s: string): string {
  return s.includes("@") ? getEmailMask(s) : getPhoneMask(s);
}
```
