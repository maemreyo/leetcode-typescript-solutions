# 138. Copy List With Random Pointer

### O(n) - TS with Map()

## Complexity

- Time: $$O(n)$$
- Space: $$O(n)$$

## Submission

Runtime: 51 ms · Beats 26.92%
Memory: 57.20 MB · Beats 74.13%

## Code

```typescript
function copyRandomList(head: _Node | null): _Node | null {
  const map = new Map<_Node, _Node>();
  const getCopy = (node: _Node | null): _Node | null =>
    node ? (map.get(node) ?? null) : null;

  let curr = head;
  while (curr) {
    map.set(curr, new _Node(curr.val));
    curr = curr.next;
  }

  curr = head;
  while (curr) {
    const clone = map.get(curr)!;
    clone.next = getCopy(curr.next);
    clone.random = getCopy(curr.random);
    curr = curr.next;
  }

  return getCopy(head);
}
```
