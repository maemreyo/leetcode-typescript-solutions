# 804. Unique Morse Code Words

### O(n) - TS with Set()

## Complexity

- Time: $$O(n)$$
- Space: $$O(n)$$

## Submission

Runtime: 2 ms · Beats 74.36%
Memory: 57.72 MB · Beats 41.03%

## Code

```typescript
const morseCode = [
  ".-",
  "-...",
  "-.-.",
  "-..",
  ".",
  "..-.",
  "--.",
  "....",
  "..",
  ".---",
  "-.-",
  ".-..",
  "--",
  "-.",
  "---",
  ".--.",
  "--.-",
  ".-.",
  "...",
  "-",
  "..-",
  "...-",
  ".--",
  "-..-",
  "-.--",
  "--..",
];

function uniqueMorseRepresentations(words: string[]): number {
  const set = new Set<string>();

  words.forEach((word) => {
    let morse = "";

    word.split("").forEach((c) => {
      const i = c.charCodeAt(0) - 97;
      const code = morseCode[i];
      morse += code;
    });

    set.add(morse);
  });

  return set.size;
}
```
