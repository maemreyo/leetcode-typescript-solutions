# 3541. Find Most Frequent Vowel And Consonant

### O(n) - TS with frequency array and Set

## Complexity

- Time: $$O(n)$$
- Space: $$O(1)$$

## Submission

- Runtime: 2 ms · Beats 91.36%
- Memory: 58.12 MB · Beats 71.60%

## Code

```typescript
const VOWEL_IDX = new Set([0, 4, 8, 14, 20]); // a,e,i,o,u

function maxFreqSum(s: string): number {
  const counts = new Array(26).fill(0);
  for (let i = 0; i < s.length; i++) {
    counts[s.charCodeAt(i) - 97]++;
  }

  let maxVowel = 0;
  let maxCons = 0;
  for (let i = 0; i < 26; i++) {
    if (VOWEL_IDX.has(i)) maxVowel = Math.max(maxVowel, counts[i]);
    else maxCons = Math.max(maxCons, counts[i]);
  }

  return maxVowel + maxCons;
}
```
