# 3264. Final Array State After K Multiplication Operations I

### O((n + k) log n) - TS with Heap

## Complexity

- Time: $$O((n + k) \log n)$$
- Space: $$O(n)$$

## Submission

Runtime: 13 ms · Beats 16.00%
Memory: 60.41 MB · Beats 24.00%

## Code

```typescript
import { Heap } from "@datastructures-js/heap";

type Entry = { i: number; value: number };

function getFinalState(
  nums: number[],
  k: number,
  multiplier: number
): number[] {
  const heap = new Heap<Entry>((a, b) => a.value - b.value || a.i - b.i);
  const res = new Array<number>(nums.length);

  nums.forEach((value, i) => {
    heap.push({ i, value });
  });

  for (let i = 0; i < k; i++) {
    const top = heap.pop()!;
    top.value *= multiplier;
    heap.push(top);
  }

  heap.toArray().forEach(({ i, value }) => {
    res[i] = value;
  });

  return res;
}
```
